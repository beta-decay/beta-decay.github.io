from PIL import Image as I width = 10000 height = 10000 im = I.new("RGB",[width,height],"white") pix = im.load() z_0 = 0 plot_size = 3.5 offset = complex(0.5,0) function = lambda z, C: z**2 + C for x in range(width): for y in range(height): z_n = z_0 a = y/height * plot_size - plot_size/2 b = x/width * plot_size - plot_size/2 C = complex(a,b) - offset colour = (0,0,0) for i in range(100): z_n = function(z_n, C) if abs(z_n) > 2: rgb_val = 255-int(255/(i+1)) colour = (255,rgb_val,rgb_val) break pix[x,y] = colour im.save("multibrot_output4.png") im.show() input("\n\nPress ENTER to continue")
